    <input type="text" placeholder="Enter book title" [(ngModel)] ="newBookTitle">
    <button (click)="addBook()">Add Book</button>
</div>

<ul>
    <li *ngFor="let book of books; let i = index">
    <!-- <span [style.textDecoration]="book.completed ? 'line-through' : 'none' ">{{book.title}}</span> -->
    <span [ngClass]="book.completed ? 'strike' : '' ">{{book.title}}</span>
    <button (click)="completeBook(book)">Completed</button>
    <button (click)="deleteBook(i)">Delete</button>
</li>

import { Component } from '@angular/core';
interface Book{
  title:string;
  completed:boolean;
}
@Component({
  selector: 'app-book-list',
  templateUrl: './book-list.component.html',
  styleUrls: ['./book-list.component.css']
})
export class BookListComponent {
  books: Book[] = [];
  newBookTitle: string = '';

  addBook(){
    const trimmedTitle = this.newBookTitle.trim();
    if(trimmedTitle){
      this.books.push({
        title: trimmedTitle,
        completed: false
      })
      this.newBookTitle='';
    }
  }

  completeBook(book: Book){
    book.completed = !book.completed;
  }

  deleteBook(index: number){
    this.books.splice(index,1);
  }
}
